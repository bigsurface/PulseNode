#include < FastLED.h >

#define LED_PIN 3
#define NUM_LEDS 200
#define BRIGHTNESS 64
#define LED_TYPE WS2811
#define COLOR_ORDER GRB
CRGB leds[NUM_LEDS];

int PulsePin = A0;
int Signal;
int Threshold = 550;

#define UPDATES_PER_SECOND 100

void setup(){
delay(3000);
FastLED.addLeds< LED_TYPE, LED_PIN, COLOR_ORDER>
(leds, NUM_LEDS).setCorrection(TypicalLEDStrip);
FastLED.setBrightness(BRIGHTNESS);
}

void loop(){
Signal = analogRead(PulsePin);
Serial.println(Signal);
if (Signal > Threshold){
leds[0] = CRGB(0,255,0);
...
leds[199] = CRGB(0,255,0);
FastLED.show();
} else {
leds[0] = CRGB(255,0,0);
...
leds[199] = CRGB(255,0,0);
FastLED.show();
}
delay(100);
}
